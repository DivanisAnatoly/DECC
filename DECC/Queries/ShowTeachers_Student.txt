SELECT CONCAT("Users"."Surname",' ',
		"Users"."Fathername",' ',
		"Users"."Name") AS "Преподаватель",
		"Users"."Birthdate" AS "Дата рождения",
		"Users"."Phone" AS "Телефон",
		"Positions"."Name" AS "Должность"
FROM PUBLIC."Users",
	"Teachers",
	"Positions"
WHERE "Teachers"."Id" = "Users"."Id"
	AND "Teachers"."PositionId" = "Positions"."Id"
